function update(worldState) { const dv = [ worldState.target.position[0] - worldState.player.position[0], worldState.target.position[1] - worldState.player.position[1], ] let mv = [0,0] mv[0] = dv[0]>0 ? 1 : dv[0]<0 ? -1 : 0 mv[1] = dv[0] ? 0 : dv[1]>0 ? 1 : dv[1]<0 ? -1 : 0 worldState.player.move(mv) }
